```R
setGeneric("f", function(x, y)x)
setMethod("f", signature(x="character", y="character"), function(x, y)"character")
f("abc", "def")

setMethod("Ops", signature(e1 = "character"), function(e1, e2) paste0(e1, e2))
Ops("abc", "def")
```
